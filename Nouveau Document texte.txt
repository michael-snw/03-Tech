@echo off

echo Starting the Git update process...

:: Naviguer vers le répertoire de base
cd "F:\AIIA\Git"

:: Boucle à travers les dossiers
for %%d in ("01 - Me" "02 - Skill" "03 - Tech" "04 - AI" "05 - Streaming") do (
    echo Processing %%d...
    cd "F:\AIIA\Git\%%d"
    
    :: Ajouter les modifications
    git add .
    
    :: Demander un message de commit
    set /p commitMessage="Enter the commit message for %%d: "
    
    :: Faire le commit avec le message
    git commit -m "%commitMessage%"
    
    :: Pousser les modifications
    git push

    :: Revenir au répertoire de base
    cd "F:\AIIA\Git"
)

echo All projects have been updated!
pause
